import pandas as pd # 假設我們有一個名為 'data.csv' 的 CSV 檔案 data = pd.read_csv('data.csv') # 在網頁上建立一個表格 table_html = '' table_html += '' for column in data.columns: table_html += f'' table_html += '' for _, row in data.iterrows(): table_html += '' for value in row: table_html += f'' table_html += '' table_html += '
{column}
{value}
' # 將表格 HTML 插入到網頁上 Element('table-container').write(table_html)
